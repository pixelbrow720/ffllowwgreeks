# Read latest dealer state for SPX 0DTE curl https://api.flowgreeks.flowjob.id/api/snapshot/spx \ -H "Authorization: Bearer $FLOWGREEKS_KEY"
// fetch SPX 0DTE snapshot const res = await fetch("https://api.flowgreeks.flowjob.id/api/snapshot/spx", { headers: { "Authorization": `Bearer ${process.env.FLOWGREEKS_KEY}` }, }); const snap = await res.json(); console.log(snap.net_gex, snap.dpi.composite);
# pip install requests import os, requests key = os.environ["FLOWGREEKS_KEY"] r = requests.get( "https://api.flowgreeks.flowjob.id/api/snapshot/spx", headers={"X-API-Key": key}, ) print(r.json()["net_gex"])
// subscribe to live state stream (ws) const ws = new WebSocket("wss://api.flowgreeks.flowjob.id/ws/live"); ws.onopen = () => ws.send(JSON.stringify({ action: "subscribe", symbols: ["spx", "ndx"], kinds: ["gex", "narrative", "alert"], })); ws.onmessage = (e) => console.log(JSON.parse(e.data));
